```python
class HardConstraintFilter:
    def filter(self, query, properties):
        """一票否决逻辑：任何硬约束不满足则排除"""
        filtered = []
        
        for prop in properties:
            # 预算硬上限（无缓冲）
            if query.budget_strict and prop.rent > query.max_budget:
                continue  # ❌ 排除
            
            # 宠物政策（严格匹配）
            if query.has_pet and not prop.allows_pets:
                continue  # ❌ 排除
            
            # 台电计费（精确匹配）
            if query.requires_standard_tpe and prop.special_tpe_rate:
                continue  # ❌ 排除
            
            filtered.append(prop)
        
        return filtered
```

```python
class OSRMClient:
    def __init__(self, osrm_server="http://router.project-osrm.org"):
        self.server = osrm_server
        self.cache = {}  # 缓存避免重复请求
    
    def get_distance(self, origin, destination, profile="walking"):
        """
        获取真实路网距离
        返回距离(米)、时间(秒)
        """
        key = (origin, destination, profile)
        if key in self.cache:
            return self.cache[key]
        
        # 调用OSRM API
        # result = osrm_api_call(...)
        # self.cache[key] = result
        # return result
        pass
    
    def compute_weight(self, origin, destination, profile="walking"):
        """
        计算0-1之间的权重（越近权重越高）
        """
        distance = self.get_distance(origin, destination, profile)
        # 归一化：距离<5km权重=1, 距离>20km权重=0
        weight = max(0, 1 - distance/20000)
        return weight
```

```javascript
class SemanticRanker {
    async rank(query, candidates) {
        // 加载ONNX模型
        const session = await ort.InferenceSession.create(modelPath);
        
        // 批量推论
        const results = [];
        for (const candidate of candidates) {
            const logits = await this.inference(session, query, candidate);
            results.push({
                property: candidate,
                score: logits[1]  // 正样本概率
            });
        }
        
        // 应用OSRM路网权重
        results.forEach(r => {
            r.final_score = r.score * r.osrm_weight;
        });
        
        // 排序返回
        return results.sort((a, b) => b.final_score - a.final_score);
    }
}
```

**影响**: 路网权重计算速度慢
